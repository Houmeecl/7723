1. Crear el Backend (Node.js):
Primero, crea un archivo llamado server.js. Este archivo se encargará de servir los contratos y manejar las solicitudes de generación de documentos.

Contenido de server.js:
javascript
Copiar
Editar
const express = require('express');
const path = require('path');
const fs = require('fs');
const app = express();

// Middleware para manejar los datos del formulario
app.use(express.json());
app.use(express.urlencoded({ extended: true }));

// Ruta para servir archivos estáticos (como los contratos en PDF o HTML)
app.use('/docs', express.static(path.join(__dirname, 'docs')));

// Ruta para obtener todos los contratos disponibles
app.get('/contratos', (req, res) => {
    const contratosPath = path.join(__dirname, 'docs', 'contratos');
    fs.readdir(contratosPath, (err, files) => {
        if (err) {
            return res.status(500).send('Error al leer los archivos de contratos.');
        }
        res.json({ contratos: files });
    });
});

// Ruta para generar el contrato con los datos del formulario
app.post('/generate-contract', (req, res) => {
    const { tipoDocumento, datosFormulario } = req.body;

    // Lógica para generar el documento con los datos recibidos
    const documentoGenerado = {
        nombre: datosFormulario.nombreComprador,
        fecha: datosFormulario.fecha,
        monto: datosFormulario.monto
    };

    res.json({
        mensaje: 'Documento generado con éxito',
        documentoGenerado: documentoGenerado
    });
});

// Configurar el puerto para que el servidor esté disponible
app.listen(3000, () => {
    console.log('Servidor corriendo en puerto 3000');
});

<!DOCTYPE html>
<html lang="es">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Formulario de Contrato</title>
</head>
<body>

    <h1>Formulario de Contrato de Compraventa</h1>

    <form id="contract-form">
        <label for="nombreComprador">Nombre del Comprador:</label>
        <input type="text" id="nombreComprador" name="nombreComprador" required><br><br>

        <label for="nombreVendedor">Nombre del Vendedor:</label>
        <input type="text" id="nombreVendedor" name="nombreVendedor" required><br><br>

        <label for="monto">Monto de la Compraventa:</label>
        <input type="number" id="monto" name="monto" required><br><br>

        <label for="fecha">Fecha de Compraventa:</label>
        <input type="date" id="fecha" name="fecha" required><br><br>

        <button type="submit">Generar Documento</button>
    </form>

    <h2>Contratos Disponibles</h2>
    <ul id="contratos-lista">
        <!-- Los contratos serán cargados aquí -->
    </ul>

    <script>
        // Cargar los contratos disponibles
        async function cargarContratos() {
            const response = await fetch('/contratos');
            const data = await response.json();
            const contratosUl = document.getElementById('contratos-lista');

            data.contratos.forEach((contrato) => {
                const contratoItem = document.createElement('li');
                contratoItem.innerHTML = `
                    <a href="/docs/contratos/${contrato}" target="_blank">${contrato}</a>
                `;
                contratosUl.appendChild(contratoItem);
            });
        }

        cargarContratos();

        // Enviar el formulario para generar el contrato
        document.getElementById('contract-form').addEventListener('submit', async (event) => {
            event.preventDefault();

            const formData = new FormData(event.target);
            const datosFormulario = {};
            formData.forEach((value, key) => {
                datosFormulario[key] = value;
            });

            const response = await fetch('/generate-contract', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ tipoDocumento: 'compraventa', datosFormulario })
            });

            const result = await response.json();
            alert(result.mensaje);
        });
    </script>

</body>
</html>
